#include <conio.h>
#include "pch.h"
#include <iostream>
using namespace std;
#define MAXSIZE 3
typedef int ElemType;
class SeQueue
{
private:
	ElemType elem[MAXSIZE];
	int front, rear,count;
public:
	SeQueue();
	~SeQueue();
	void Display();
	void AddQ(ElemType x);
	ElemType DelQ();
};
SeQueue::SeQueue()
{
	front = 0;
	rear = 0;
	count = 0;
	cout << "init!" << endl;
}
SeQueue::~SeQueue()
{};//{ delete [MAXSIZE]Q.elem;}
void SeQueue::Display()//遍历队列
{
	ElemType x; int j = 0;
	if (count==0)
	{
		cout << "QUEUE IS FULL!";
	}
	else {
		for (ElemType e : elem) {
			if (e>=0) {
				cout << e << " ";
			}
			
		}
	}
	cout << endl;
}
void SeQueue::AddQ(ElemType x)//入队
{
	if (count== MAXSIZE) {
		cout << "队列已满！" << endl;
	}
	else {
		elem[rear] = x;
		rear = (rear + 1) % MAXSIZE;
		cout << "入队成功！" << endl;
		count++;
	}
}
ElemType SeQueue::DelQ()//出队
{
	if (count==0) {
		cout << "队列已空！" << endl;
	}else{
		ElemType e;
		e = elem[front];
		elem[front] = -1;
		front = (front + 1) % MAXSIZE;
		count--;
		return e;
	}
}

	ElemType e;
	SeQueue h;

	cout << "\n                           队列存储结构演示";

		cout << "\n******************************** ";
		cout << "\n    请输入你的选择(1,2,3,4,5)";
		cin >> k;
		switch (k) {
		case 1: {SeQueue::SeQueue();
		}break;
		case 2: {h.Display();
		}break;
		case 3: {
			cout << "进队  data=?";  cin >> e;
			h.AddQ(e);
			h.Display();
		}break;
		case 4: {   e = h.DelQ();
			if (e != -1)
