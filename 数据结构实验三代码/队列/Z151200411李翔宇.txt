#include <iostream>
#include <conio.h>
#define MAXSIZE 3
using namespace std;
typedef int ElemType;
 class SeQueue
  {  private:
       ElemType elem[MAXSIZE];
        int front,rear;
     public:
        SeQueue();
        ~SeQueue();
        void Display();
        void AddQ(ElemType x);
        ElemType DelQ();
		int getfr()
		{
			return front;
		}
		int getre()
		{
			return rear;
		}
};
SeQueue::SeQueue()
  {  front=0;
     rear=0;
     cout<<"init!"<<endl;
  }
SeQueue::~SeQueue()
{};//{ delete [MAXSIZE]Q.elem;}
void SeQueue::Display()
{ 
	ElemType x; 
	int j = 0;
	if(rear	== front)
	{	
		cout<<"QUEUE IS EMPTY!";
	}
	else
	{
		j =(front + 1)%MAXSIZE;
		while(j!=(rear+1) %MAXSIZE)
	{
		x = elem[j - 1];
		cout<< x <<" ";
		j = (j+1)%MAXSIZE;
	}
}
	cout<<endl;
}
void SeQueue::AddQ(ElemType x)
{  
	if((rear+1) % MAXSIZE == front)
	{
		cout << "队满" << endl;
	}
	else {
		elem[rear] = x;
		rear = (rear + 1) % MAXSIZE;
		cout << "成功" << endl;
	}
		//将此成员函数补充完整，使得程序能够正确运行
 }
 ElemType SeQueue::DelQ()
 {
	 ElemType e;
	 if(rear == front)
	 {
		 return -1;
	 }
	 else
	 {
		 e = elem [front];
		 front = (front + 1) % MAXSIZE;
		 return e;
	 }
 //将此成员函数补充完整，使得程序能够正确运行
 }
