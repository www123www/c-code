void SeQueue::AddQ(ElemType x)

{  //将此成员函数补充完整，使得程序能够正确运行
	if ((rear + 1) % MAXSIZE == front)
		cout << "over_flow" << endl;
	else
	{    
		rear = (rear + 1) % MAXSIZE;
		elem[rear] = x;
		cout << "SUCCESS" << endl;
	}
 }

 {

	 ElemType e;

		cout << "under_flow" << endl;
		return -1;
