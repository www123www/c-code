#include <iostream>
using namespace std;
const int MAXSIZE=5; //最大值为5
class Sqlist
{
private:
	int  elem[MAXSIZE];
	int length;
public:
		Sqlist( void);
		~Sqlist(){ };
		void  SetData();
		void  Insert( int e);
		void Delet(int i);
		void PrintOut();
};
void Sqlist::PrintOut()  //输出
 { 
	
	
	cout<<"排序"<<endl;
	for(int k=0; k<length;k++)      
	{ 
		cout<<elem[k]<<"  ";
	}
 }
Sqlist::Sqlist( ) { length=0;}

{
	cout<<"\n 输入个数length="<<endl;
	cin>>length;
	int a,b;
	bool c=0;
    if(length<= MAXSIZE)
	{	cout<<"\n 请从小到大输入"<<endl;
		for(int i=0;i<length;i++)
		{

			cin>>a;
			if(c==0)
			{
			  c=1;
			  b=a;
			  elem[i]=a;
				
			}else if(b<=a)
			{ 
				elem[i]=a;
			
			}else

				    i--;
			} 
				  

			
			
		}

}
void  Sqlist::Insert( int e) //定值插入
{
	

	if(length==MAXSIZE)  //判断是否到达最大值
	{
		cout<<"通讯录位置已满"<<endl;
	  return ;
	}else 
	{
	for(int j=0;j<length;j++)     //从第1个开始查找
		
		if(e<elem[j])         //判断e是否小于elem[j]  
		{
		  for(int k=length;k>=j+1;k--)   //是就将e插入到elem[j]，elem[j]到elem[length]的数位置向后移动变成elem[j+1]到elem[length+1]
		  {
		     elem[k]=elem[k-1];
		  }elem[j]=e;
			length++;       //计数加1
			return;
		}
	}
	elem[length]=e;   //插入的数大于顺序表中所有的数，直接插入在最后面；
	length++;
}
void Sqlist::Delet(int e)//删除

{
	
	
	

	if(length<=0)
	
	{	
		cout<<"顺序表不存在内容 "<<endl;//判断链为不为空
	    return ;
	}
	else{
		
	 for(int j=0;j<length;j++)//从第1个开始查找
	 {
	    if(elem[j]==e)  //判断要删除的数
	    {	
		for(int k=j;k<length;k++)  //是进行删除，
		elem[k]=elem[k+1];
		length--;
		j--;      //删除后j位置的值改变，需要从新再次判断。
		}
	 }
	}
}
